import fused

@fused.udf
def udf():
    import geopandas as gpd

    # Shape file as zip
    url="https://www2.census.gov/geo/tiger/TIGER_RD18/STATE/11_DISTRICT_OF_COLUMBIA/11/tl_rd22_11_bg.zip"
    gdf = gpd.read_file(url)
    return gdf

gdf = fused.run(udf=udf, engine="local")
