@fused.udf
def udf(bbox: fused.types.TileGDF=None, n: int=10):
    import geopandas as gpd
    from shapely.geometry import Polygon
    
    # Create example polygons
    polygon1 = Polygon([(-123.0, 37.6), (-122.4, 37.6), (-122.4, 37.9), (-123.0, 37.9)])
    polygon2 = Polygon([(-122.6, 37.7), (-122.0, 37.7), (-122.0, 38.0), (-122.6, 38.0)])
    
    # Create a GeoDataFrame
    gdf = gpd.GeoDataFrame({'geometry': [polygon1, polygon2], 'category': ['A', 'A']})
    
    # Dissolve the GeoDataFrame by 'category'
    dissolved_gdf = gdf.dissolve(by='category')
    
    # Display the result
    return dissolved_gdf
