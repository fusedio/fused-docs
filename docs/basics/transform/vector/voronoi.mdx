# Voronoi

```python
@fused.udf
def udf(object_to_show: str = 'voronoi', n: int=100):
    import geopandas as gpd
    import numpy as np
    import matplotlib.pyplot as plt
    from shapely.geometry import Point, Polygon
    from scipy.spatial import Voronoi, voronoi_plot_2d

    minx, miny, maxx, maxy = -5.97468,35.5492, -5.37391,35.89908

    np.random.seed(2)
    # Generate random points
    points = np.random.rand(n, 2)  # 10 random points in the unit square
    points[:, 0] = points[:, 0] * (maxx - minx) + minx
    points[:, 1] = points[:, 1] * (maxy - miny) + miny

    # Create Voronoi diagram
    vor = Voronoi(points)
    
    # Extract Voronoi regions and vertices
    regions, vertices = vor.regions, vor.vertices
    
    # Create polygons from Voronoi regions
    polygons = []
    for region in regions:
        if not -1 in region:
            polygon = Polygon(vertices[region])
            polygons.append(polygon)
    
    # Create a GeoDataFrame
    if object_to_show == 'voronoi':
        return gpd.GeoDataFrame({'geometry': polygons})
    else:
        return gpd.GeoDataFrame({'geometry': [Point(p) for p in points]})
```
